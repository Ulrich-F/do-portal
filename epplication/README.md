# Docker

## requirements
 1) install docker
 2) install docker-compose

## installation
git clone https://github.com/certat/do-portal.git

## run do-portal
cd do-portal/backend
docker-compose up

## run cp-portal
cd do-portal/frontend
docker-compose up

## run epplication
cd do-portal/epplication
docker-compose up

## run ui-tests
cd do-portal/epplication
bash test.sh
